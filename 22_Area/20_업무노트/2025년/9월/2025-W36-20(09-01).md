#### ▣ 완료된 작업
```dataviewjs
const current = dv.current();

// 파일명에서 연도와 주차 추출 (예: 2025-W33-20 -> year: 2025, week: 33)
const fileName = current.file.name;
const weekMatch = fileName.match(/(\d{4})-W(\d+)/);

if (weekMatch) {
    const currentYear = parseInt(weekMatch[1]);
    const currentWeek = parseInt(weekMatch[2]);
    
    // 전주 계산 (현재 주차 - 1)
    const prevWeek = currentWeek - 1;
    const prevYear = prevWeek < 1 ? currentYear - 1 : currentYear;
    const targetWeek = prevWeek < 1 ? 52 : prevWeek;

    // 전주 월요일부터 금요일까지의 일간노트 찾기
    const pages = dv.pages('"20_Marketing_Team/22_Area/20_업무노트"')
        .where(p => {
            if (!p.file.day) return false;
            const fileWeek = p.file.day.weekNumber;
            const fileYear = p.file.day.weekYear;
            const dayOfWeek = p.file.day.weekday; // 1=월요일, 7=일요일
            
            return fileYear === prevYear && 
                   fileWeek === targetWeek && 
                   dayOfWeek >= 1 && dayOfWeek <= 5; // 월~금만
        });

    // 완료된 작업들 수집
    const completedTasks = [];
    for (let page of pages) {
        const tasks = page.file.tasks.where(t => t.completed);
        for (let task of tasks) {
            completedTasks.push({
                date: page.file.day.toFormat("MM-dd(ccc)"),
                task: task.text,
                file: page.file.link
            });
        }
    }

    if (completedTasks.length > 0) {
        // 날짜별로 그룹화
        const grouped = completedTasks.reduce((acc, item) => {
            if (!acc[item.date]) {
                acc[item.date] = {
                    tasks: [],
                    fileLink: item.file
                };
            }
            acc[item.date].tasks.push(item.task);
            return acc;
        }, {});

        dv.paragraph(`대상 기간: ${prevYear}년 ${targetWeek}주차`);
        
        for (let [date, data] of Object.entries(grouped).sort()) {
            dv.paragraph(`**${date} ${data.fileLink}**`);
            for (let task of data.tasks) {
                dv.paragraph(`- ${task}`);
            }
        }
    } else {
        dv.paragraph(`대상 기간: ${prevYear}년 ${targetWeek}주차`);
        dv.paragraph("전주에 완료된 작업이 없습니다.");
    }
} else {
    dv.paragraph("파일명이 'YYYY-W##' 형식이 아닙니다. (예: 2025-W33-20)");
}
```
